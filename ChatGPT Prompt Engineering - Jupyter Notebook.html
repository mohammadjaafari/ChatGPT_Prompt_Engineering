    import os

    from dotenv import load_dotenv, find_dotenv
    _ = load_dotenv(find_dotenv()) # read local .env file
    ​
    openai.api_key  = os.getenv('OPENAI_API_KEY') # the key used here is from DeepLearning.AI offered using its courses
                                                  # please substitute with your own key
    ​
    def get_completion(prompt, model="gpt-3.5-turbo"): # calling onto gpt 3.5 model to create completion to the message
        messages = [{"role": "user", "content": prompt}] # due to the nature of gpt model being a decoder only transformer model
        response = openai.ChatCompletion.create(
            model=model,
            messages=messages,
            temperature=0, # this is the degree of randomness of the model's output
        )
        return response.choices[0].message["content"] # returning its message

    def get_completion_from_messages(messages, model="gpt-3.5-turbo", temperature=0):
        response = openai.ChatCompletion.create(
            model=model,
            messages=messages,
            temperature=temperature, # this is the degree of randomness of the model's output
        )
        return response.choices[0].message["content"]
